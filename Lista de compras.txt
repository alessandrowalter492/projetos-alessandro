lista = []

print (''' =========== LISTA DE COMPRAS SANDRANNE ===========

1 - Adicionar item
2 - Ver lista
3 - Remover item
0 - Sair''')

while True:
    escolha = int(input('Qual opção deseja? '))
    if escolha == 1:
        add = input ('Qual item deseja adicionar? ')
        lista.append(add)
        print (f'Esta é sua lista atual {lista}')
    elif escolha == 2:
        for listas in lista:
            print (listas)
        if len (lista) == 0:
            print ('Sua lista esta vazia')
    elif escolha == 3:
        remover = input ('Qual item deseja remover? ')
        if remover not in lista:
            print ('Este item nao existe')

            continue
        lista.remove (remover)
        print (f'{remover} foi removido com sucesso')
        print (lista)
    elif escolha == 0:
        print ('Ate a proxima')

        break
    else:
        print ('Escolha uma opção valida')
