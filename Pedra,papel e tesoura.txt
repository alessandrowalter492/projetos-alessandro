import random

valor_aleatorio = random.randint(1, 3)

print('Vamos jogar pedra, papel ou tesoura?')

escolha = int(input('''1 = Pedra
2 = Papel
3 = Tesoura

Qual você escolhe? '''))

while escolha not in (1, 2, 3):
    print('Você deve escolher uma das opções válidas.\n')

    escolha = int(input('''1 = Pedra
2 = Papel
3 = Tesoura

Qual você escolhe? '''))

if escolha == 1:
    jogada = 'Pedra'
elif escolha == 2:
    jogada = 'Papel'
else:
    jogada = 'Tesoura'

if valor_aleatorio == 1:
    pc = 'Pedra'
elif valor_aleatorio == 2:
    pc = 'Papel'
else:
    pc = 'Tesoura'

print(f'\nVocê escolheu {jogada}')
print(f'O computador escolheu {pc}')

if escolha == valor_aleatorio:
    print('\nEmpate!')

elif (
    (escolha == 1 and valor_aleatorio == 3) or
    (escolha == 2 and valor_aleatorio == 1) or
    (escolha == 3 and valor_aleatorio == 2)
):
    print('\n🎉 Você venceu!')

else:
    print('\n💻 O computador venceu!')
