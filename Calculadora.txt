
while True:

    print ('''      Calculadora

1 - Somar
2 - Subtrair
3 - Multiplicar
4 - Dividir
0 - Sair

''')

    escolha = int(input('Escolha uma das opçoes: '))
    if escolha == 0:
            print ('Volte sempre')

            break


    while escolha not in (1,2,3,4,0):

        print ('Escolha uma opção válida ')

        escolha = int(input('Escolha uma das opçoes: '))

        


    pvalor = int(input('Digite o primeiro valor: '))
    svalor = int(input('Digite o segundo valor: '))

    if escolha == 1:
        print (f'O valor da sua soma é {pvalor + svalor}')
    elif escolha == 2:
        print (f'O valor da sua subtração é {pvalor - svalor}')
    elif escolha == 3:
        print (f'o valor da sua multiplicação é {pvalor * svalor}')
    elif escolha == 4:
        print (f'o valor da sua divisão é {pvalor / svalor}')
