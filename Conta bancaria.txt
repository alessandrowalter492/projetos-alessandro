saldo = 1000
print (''''=======Bem vindo a sua conta bancaria=======
    
       Agora selecione a opção desejada
    
1 - Ver saldo
2 - Depositar
3 - Sacar
0 - sair''')

while True:
    escolha = int(input('Qual opção deseja? '))

    if escolha == 1:
        print (f'Seu saldo é {saldo}')
    elif escolha == 2:
        deposito = int(input('Qual valor deseja depositar? '))
        saldo = saldo + deposito
        print (f'Seu saldo agora é de {saldo}')
    elif escolha == 3:
        
        saque = int(input('Qual valor deseja sacar? '))
        if saque > saldo:
            print ('Saldo insuficiente')

            break
        saldo = saldo - saque
        print (f'Seu saldo agora é de {saldo}')
    elif escolha == 0:
        print ('Volte sempre!')
        break
    
    else:
        print('Escolha uma opção valida!')
